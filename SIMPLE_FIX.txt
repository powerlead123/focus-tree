问题：drawBullets函数（780-1169行）中的代码试图访问bullet.trail, bullet.particles等不存在的数组，导致错误和卡顿。

解决方案：用以下简化代码替换reading.js中的drawBullets函数（780-1169行）：

```javascript
function drawBullets() {
    bullets.forEach(bullet => {
        ctx.save();
        
        const size = bullet.size || 5;
        const x = bullet.x;
        const y = bullet.y;
        
        // 简化版特效 - 无粒子系统
        switch(bullet.type) {
            case 'fire':
                ctx.fillStyle = '#ff6600';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'ice':
                ctx.fillStyle = '#00ffff';
                ctx.strokeStyle = '#ffffff';
                ctx.lineWidth = 1;
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                ctx.stroke();
                break;
                
            case 'lightning':
                const zigzag = Math.sin(bullet.age * 0.3) * 3;
                ctx.strokeStyle = '#ffff00';
                ctx.lineWidth = size;
                ctx.beginPath();
                ctx.moveTo(x + zigzag, y - 8);
                ctx.lineTo(x - zigzag, y + 8);
                ctx.stroke();
                break;
                
            case 'poison':
                const pulse = Math.sin(bullet.age * 0.2) * 2;
                ctx.fillStyle = 'rgba(0, 255, 0, 0.3)';
                ctx.beginPath();
                ctx.arc(x, y, size + pulse, 0, Math.PI * 2);
                ctx.fill();
                ctx.fillStyle = '#00ff00';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'crystal':
                ctx.translate(x, y);
                ctx.rotate(bullet.angle);
                ctx.fillStyle = '#ff00ff';
                ctx.strokeStyle = '#ffffff';
                ctx.lineWidth = 1;
                ctx.beginPath();
                for (let i = 0; i < 6; i++) {
                    const a = (Math.PI * 2 / 6) * i;
                    const px = Math.cos(a) * size;
                    const py = Math.sin(a) * size;
                    if (i === 0) ctx.moveTo(px, py);
                    else ctx.lineTo(px, py);
                }
                ctx.closePath();
                ctx.fill();
                ctx.stroke();
                break;
                
            case 'wind':
                const wave = Math.sin(bullet.age * 0.2) * 4;
                ctx.strokeStyle = '#87ceeb';
                ctx.lineWidth = size;
                ctx.beginPath();
                ctx.moveTo(x + wave, y - 6);
                ctx.quadraticCurveTo(x, y, x - wave, y + 6);
                ctx.stroke();
                break;
                
            case 'rainbow':
                ctx.fillStyle = `hsl(${bullet.hue}, 100%, 50%)`;
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'star':
                ctx.fillStyle = '#ffd700';
                ctx.translate(x, y);
                ctx.rotate(bullet.angle);
                ctx.beginPath();
                for (let i = 0; i < 5; i++) {
                    const a = (Math.PI * 2 / 5) * i - Math.PI / 2;
                    const px = Math.cos(a) * size;
                    const py = Math.sin(a) * size;
                    if (i === 0) ctx.moveTo(px, py);
                    else ctx.lineTo(px, py);
                    const ia = a + Math.PI / 5;
                    const ipx = Math.cos(ia) * size * 0.4;
                    const ipy = Math.sin(ia) * size * 0.4;
                    ctx.lineTo(ipx, ipy);
                }
                ctx.closePath();
                ctx.fill();
                break;
                
            case 'holy':
                ctx.strokeStyle = '#ffffff';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(x - size * 1.2, y);
                ctx.lineTo(x + size * 1.2, y);
                ctx.moveTo(x, y - size * 1.2);
                ctx.lineTo(x, y + size * 1.2);
                ctx.stroke();
                ctx.fillStyle = '#ffffff';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'water':
                const ripple = (bullet.age * 2) % 12;
                ctx.strokeStyle = `rgba(30, 144, 255, ${1 - ripple / 12})`;
                ctx.lineWidth = 1;
                ctx.beginPath();
                ctx.arc(x, y, ripple, 0, Math.PI * 2);
                ctx.stroke();
                ctx.fillStyle = '#1e90ff';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'gold':
                ctx.fillStyle = '#ffd700';
                ctx.translate(x, y);
                ctx.rotate(bullet.angle);
                ctx.beginPath();
                ctx.moveTo(0, -size);
                ctx.lineTo(size, 0);
                ctx.lineTo(0, size);
                ctx.lineTo(-size, 0);
                ctx.closePath();
                ctx.fill();
                break;
                
            case 'firework':
                ctx.fillStyle = '#ff1493';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'nebula':
                ctx.fillStyle = 'rgba(147, 112, 219, 0.6)';
                ctx.beginPath();
                ctx.arc(x, y, size * 1.3, 0, Math.PI * 2);
                ctx.fill();
                ctx.fillStyle = '#9370db';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'laser':
                ctx.strokeStyle = '#ff0000';
                ctx.lineWidth = size;
                ctx.beginPath();
                ctx.moveTo(x, y + 15);
                ctx.lineTo(x, y - 15);
                ctx.stroke();
                ctx.strokeStyle = '#ffffff';
                ctx.lineWidth = size * 0.3;
                ctx.stroke();
                break;
                
            case 'nuclear':
                ctx.fillStyle = '#00ff00';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'meteor':
                ctx.fillStyle = 'rgba(255, 140, 0, 0.4)';
                ctx.beginPath();
                ctx.arc(x, y + 8, size * 0.7, 0, Math.PI * 2);
                ctx.fill();
                ctx.fillStyle = '#ffffff';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
                break;
                
            case 'ultimate':
                const hue = (bullet.age * 10) % 360;
                ctx.translate(x, y);
                ctx.rotate(bullet.angle);
                for (let i = 0; i < 8; i++) {
                    const a = (Math.PI * 2 / 8) * i;
                    ctx.fillStyle = `hsl(${(hue + i * 45) % 360}, 100%, 50%)`;
                    ctx.beginPath();
                    ctx.arc(0, 0, size, a, a + Math.PI / 4);
                    ctx.lineTo(0, 0);
                    ctx.fill();
                }
                break;
                
            default:
                ctx.fillStyle = '#ffffff';
                ctx.beginPath();
                ctx.arc(x, y, size, 0, Math.PI * 2);
                ctx.fill();
        }
        
        ctx.restore();
        
        // 巨弹光晕
        if (bullet.size > 5) {
            ctx.strokeStyle = 'rgba(255, 255, 255, 0.4)';
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.arc(x, y, bullet.size + 3, 0, Math.PI * 2);
            ctx.stroke();
        }
        
        // 穿透标记
        if (bullet.pierce) {
            ctx.fillStyle = '#ffffff';
            ctx.font = '10px Arial';
            ctx.textAlign = 'center';
            ctx.fillText('⚡', x, y + 3);
        }
    });
}
```

- 移除了所有shadowBlur（性能杀手）
- 简化了渐变效果
- 保留了基本的视觉区分
